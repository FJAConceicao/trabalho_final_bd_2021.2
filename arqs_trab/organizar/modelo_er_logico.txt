User(id, username, city, province):
 PK(id)

Product(id, asin, brand, sourceUrl, descriptions, dimension, weight, idManufacturer):
 PK(id)
 FK(idManufacturer) references Manufacturer(id)

Manufacturer(id, name, url):
 PK(id)

Store(id, name, url):
 PK(id)

// Relaciona Produto com Store
Sells(idProduct, idStore):
 FK(idProduct) references Product(id)
 FK(idStore) references Store(id)
 PK(idProduct, idStore)

// Relaciona Produto com User
Review(id, text, title, rating, didPurchase, date, idProduct, idUser):
 FK(idProduct) references Product(id)
 FK(idUser) references User(id)
 PK(idProduct, idUser)

// Categorias multivaloradas de Produto
Categorie(categorie, idProduct)
 FK(idProduct) references Product(id)
 PK(categorie, idProduct)
